Friend Timezones

See what time it is for everyone.

The clocks update automatically.
const friends = [ { name:"Fami", utc:3 }, { name:"Annie", utc:3 }, { name:"Maow", utc:1 }, { name:"Noelle", utc:-4 }, { name:"Bing", utc:-4 } ]; ]; const friendList = document.getElementById("friend-list"); function getHourInTimezone(date, timeZone) { const hour = new Intl.DateTimeFormat("en-US", { timeZone: timeZone, hour: "2-digit", hourCycle: "h23" }).format(date); return Number(hour); } function getStatus(hour) { if (hour >= 24 || hour < 7) { return "Perhaps sleeping?"; } if (hour < 9) { return "Good morning!"; } if (hour < 18) { return "Maybe free?"; } if (hour < 22) { return "Good evening!"; } return "Getting late"; } function createFriendCard(friend, now) { const time = new Intl.DateTimeFormat("en-US", { timeZone: friend.timeZone, hour: "numeric", minute: "2-digit", second: "2-digit" }).format(now); const date = new Intl.DateTimeFormat("en-US", { timeZone: friend.timeZone, weekday: "long", month: "long", day: "numeric" }).format(now); const localHour = getHourInTimezone( now, friend.timeZone ); const status = getStatus(localHour); return `

${friend.name}

${friend.location}

${time}

${date}

${status}
`; } function updateClocks() { const now = new Date(); friendList.innerHTML = friends .map(friend => createFriendCard(friend, now)) .join(""); } updateClocks(); setInterval(updateClocks, 1000);
